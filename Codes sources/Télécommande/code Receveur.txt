#include <Arduino.h>
#include <SPI.h>
#include <nRF24L01.h>
#include <RF24.h>
#include <RF24_config.h>

 
RF24 radio(14,32); //pin CE, CSN
const byte  address[6] = "00001"; //channel d'écriture/lecture
uint16_t joyStickInput[4] = {0, 0, 0, 0};
uint16_t joyStickLeftX;
uint16_t joyStickLeftY;
uint16_t joyStickRightX;
uint16_t joyStickRightY;

void setupReception()
{
  radio.begin();
  radio.openReadingPipe(0, address);
  radio.setPALevel(RF24_PA_MIN);
  radio.startListening();
}

void setup() {
  Serial.begin(9600);
  setupReception();
}

void acquisitionInput()
{
  radio.read(joyStickInput, sizeof(joyStickInput));
  for(int i = 0; i<=3; i++) //recuperation des données de l'émetteur
                            //et mapping des valeurs pour réutilisation de celles-ci
  {
    joyStickInput[i] = map(joyStickInput[i], 0, 1023, 0, 510); 
  }
}

void affichageInput()
{
  joyStickLeftX = joyStickInput[0];
  joyStickLeftY = joyStickInput[1];
  joyStickRightX = joyStickInput[2];
  joyStickRightY = joyStickInput[3];
  Serial.print("Joystick 1 X : ");
  Serial.print(joyStickLeftX);
  Serial.print("  Joystick 2 X : ");
  Serial.println(joyStickRightX);
  Serial.print("JoyStick 1 Y : ");
  Serial.print(joyStickLeftY);
  Serial.print("  JoyStick 2 Y : ");
  Serial.println(joyStickRightY);
  Serial.println("");
}

void loopReception()
{
  acquisitionInput();
  affichageInput();
}
void loop() {
    loopReception();
}
